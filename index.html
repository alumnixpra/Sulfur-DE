INSTALL_SCRIPT='#!/usr/bin/env bash
# setup-de.sh — instala y configura el DE ultraligero
# uso: bash setup-de.sh [debian|arch|fedora|void]
set -e
DISTRO=${1:-debian}

echo ">> instalando paquetes ($DISTRO)..."
case $DISTRO in
  debian|ubuntu)
    sudo apt install -y xorg xinit i3 i3status dmenu st dunst alsa-utils network-manager ;;
  arch)
    sudo pacman -S --noconfirm xorg-server xorg-xinit i3-wm i3status dmenu st dunst alsa-utils networkmanager
    sudo systemctl enable NetworkManager ;;
  fedora)
    sudo dnf install -y xorg-x11-server-Xorg xorg-x11-xinit i3 i3status dmenu st dunst alsa-utils NetworkManager ;;
  void)
    sudo xbps-install -y xorg-server xinit i3 i3status dmenu st dunst alsa-utils NetworkManager ;;
esac

echo ">> creando directorios de config..."
mkdir -p ~/.config/i3 ~/.config/i3status ~/.config/dunst

echo ">> escribiendo ~/.xinitrc..."
cat > ~/.xinitrc << XINITRC
#!/bin/sh
xsetroot -solid black &
dunst &
exec i3
XINITRC

echo ">> escribiendo ~/.bash_profile (startx auto en tty1)..."
grep -q "startx" ~/.bash_profile 2>/dev/null || cat >> ~/.bash_profile << BPROFILE

[ -z "\$DISPLAY" ] && [ "\$(tty)" = "/dev/tty1" ] && exec startx
BPROFILE

echo ">> listo. reinicia y loguea en tty1."
echo "   atajos: Mod+Enter=terminal  Mod+d=apps  Mod+Shift+q=cerrar"
'

TROUBLESHOOT='# X no arranca:
startx -- :0 2>&1 | tail -50          # ver error completo
cat /var/log/Xorg.0.log | grep EE     # solo errores
ls /tmp/.X*                           # sockets X activos (conflicto si existen)

# i3 no carga config:
i3 --validate                         # validar config antes de recargar

# sin sonido con alsa:
cat /proc/asound/cards                # listar tarjetas
amixer -c 0 sset Master unmute        # unmute tarjeta 0

# red no conecta:
sudo systemctl status NetworkManager  # estado del servicio
sudo systemctl restart NetworkManager # reiniciar
journalctl -u NetworkManager -n 30    # ultimos logs

# pantalla negra al arrancar:
# verificar que ~/.xinitrc existe y tiene permisos de ejecucion
chmod +x ~/.xinitrc
# verificar que startx esta instalado:
which startx
'
